3 Local variable definition:

(let ((foo 4)
      (bar 5))
  (print (+ foo bar)))

4 Equality operations:

(equal 3.0 (* 6 0.5)) ;t
(equal 3 (* 6 0.5)) ;nil
(= 3 (* 6 0.5)) ;t
(null '()) ;t - check for nil

5 List operations:

(setq my-list '(3 4))
(cons 5 my-list) ;returns (5 3 4), my-list is still (3 4)
(add-to-list 'my-list 5) ;modifies my-list to (5 3 4)
(car my-list) ;returns first element of my-list
(cdr my-list) ;returns list without of first element
(nth 2 my-list) ;returns element at nth index
(member 3 my-list) ;returns nil or sub-list with value at 1st index

6 Conditionals:

(if (null '())
  (print "True. Empty list is equivalent to nil") ;true case
  (print "This won't get printed")) ;false case

(when (member 4 '(3 4))
  (print "4 is a member of the list")) ;only true case or returns nil

(unless (member 5 '(3 4))
  (print "5 is not a member"))

7 Looping

(let ((a-list '(3 4 5 8)))
  (dolist (el a-list)
    (print el)))

8 "Interactive" elisp function

An interactive elisp function is one which you can call from an emacs buffer using M-x <function> or SPC SPC <function>. Just add (interactive) at the beginning of the function to enable this feature. Here's an example that counts the number of words in the buffer.

(defun cheap-count-words ()
  (interactive)
  (save-excursion ;restore pointer location after running the commands within
    (let ((num-words 0))
      (goto-char (point-min))
      (while (forward-word)
        (setq num-words (1+ num-words)))
      (print (format "Num of words: %s" num-words))
      num-words)))
